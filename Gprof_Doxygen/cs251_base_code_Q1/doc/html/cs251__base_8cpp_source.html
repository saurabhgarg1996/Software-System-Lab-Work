53 void base_sim_t::pre_solve(b2Contact* contact, const b2Manifold* oldManifold)
54 {
55  const b2Manifold* manifold = contact->GetManifold();
56 
57  if (manifold->pointCount == 0)
58  {
59  return;
60  }
61 
62  b2Fixture* fixtureA = contact->GetFixtureA();
63  b2Fixture* fixtureB = contact->GetFixtureB();
64 
65  b2PointState state1[b2_maxManifoldPoints], state2[b2_maxManifoldPoints];
66  b2GetPointStates(state1, state2, oldManifold, manifold);
67 
68  b2WorldManifold world_manifold;
69  contact->GetWorldManifold(&world_manifold);
70 
71  for (int32 i = 0; i < manifold->pointCount && m_point_count < k_max_contact_points; ++i)
72  {
73  contact_point_t* cp = m_points + m_point_count;
74  cp->fixtureA = fixtureA;
75  cp->fixtureB = fixtureB;
76  cp->position = world_manifold.points[i];
77  cp->normal = world_manifold.normal;
78  cp->state = state2[i];
79  ++m_point_count;
80  }
81 }
